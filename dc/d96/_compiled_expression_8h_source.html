54 const MinuitParameterSet* mps = nullptr;

56 {
57 if constexpr( std::is_convertible<decltype(arg), DebugSymbols>::value ) this->m_db = arg;
58 else if constexpr( std::is_convertible<decltype(arg), std::map<std::string, unsigned>>::value ) this->m_evtMap = arg;
59 else if constexpr( std::is_convertible<decltype(arg), const MinuitParameterSet*>::value or
60 std::is_convertible<decltype(arg), const AmpGen::MinuitParameterSet*>::value or
61 std::is_convertible<decltype(arg), MinuitParameterSet*>::value ){
62 mps = arg;
63 }
64 else if constexpr( std::is_convertible<decltype(arg), MinuitParameterSet>::value ) mps = &arg;
65
66 else if constexpr( std::is_convertible<decltype(arg), disableBatch>::value ) {
67 DEBUG("Disabling bulk evaluation: did you do this on purpose?");
68 m_disableBatch = true;
69 }
70 else ERROR("Unrecognised argument: " << type_string(arg) );
71 };
72 for_each( std::tuple<const namedArgs&...>(args...), process_argument);
73 if( mps == nullptr ){ DEBUG("No minuit parameterset linked."); }

80 if( is<TensorExpression>(expression) ) m_outputSize *= cast<TensorExpression>(expression).tensor().size();
